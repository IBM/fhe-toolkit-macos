211 // Class used to multiply an encrypted row std::vector by a 1D linear
212 // transformation.
213 class MatMul1DExec : public MatMulExecBase
214 {
215 public:

217 
218  long dim;
219  long D;
220  bool native;
221  bool minimal;
222  long g;
223 
224  ConstMultiplierCache cache;
225  ConstMultiplierCache cache1; // only for non-native dimension
226 
227  // The constructor encodes all the constants for a given
228  // matrix in zzX format.
229  // The mat argument defines the entries of the matrix.
230  // Use the upgrade method (below) to convert to DoubleCRT format.
231  // If the minimal flag is set to true, a strategy that relies
232  // on a minimal number of key switching matrices will be used;
233  // this is intended for use in conjunction with the
234  // addMinimal{1D,Frb}Matrices routines declared in helib.h.
235  // If the minimal flag is false, it is best to use the
236  // addSome{1D,Frb}Matrices routines declared in helib.h.
237  explicit MatMul1DExec(const MatMul1D& mat, bool minimal = false);
238 
239  // Replaces an encryption of row std::vector v by encryption of v*mat
240  void mul(Ctxt& ctxt) const override;
241 
242  // Upgrades encoded constants from zzX to DoubleCRT.
243  void upgrade() override
244  {
245  cache.upgrade(ea.getContext());
246  cache1.upgrade(ea.getContext());
247  }
248 
249  const EncryptedArray& getEA() const override { return ea; }
