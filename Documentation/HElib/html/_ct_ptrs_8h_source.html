59 // Find the lowest level among many ciphertexts
60 // FIXME: using bitCapacity isn't really the right thing...
61 // this could break some code
62 inline long findMinBitCapacity(const CtPtrs& v)
63 {
64  long lvl = LONG_MAX;
65  for (long i = 0; i < v.size(); i++)
66  if (v.isSet(i) && !v[i]->isEmpty())
67  lvl = std::min(lvl, v[i]->bitCapacity());
68  return lvl;
69 }
70 inline long findMinBitCapacity(const CtPtrMat& m)
71 {
72  long lvl = LONG_MAX;
73  for (long i = 0; i < m.size(); i++)
74  lvl = std::min(lvl, findMinBitCapacity(m[i]));
75  return lvl;
76 }
